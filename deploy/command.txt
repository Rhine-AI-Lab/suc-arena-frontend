
# 启动
sudo systemctl start suc-arena-web.service

# 停止
sudo systemctl stop suc-arena-web.service

# 重启
sudo systemctl restart suc-arena-web.service

# 查看状态
sudo systemctl status suc-arena-web.service

# 设为开机启动
sudo systemctl enable suc-arena-web.service
